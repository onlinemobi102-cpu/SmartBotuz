import json

def load_portfolio():
    try:
        with open('data/portfolio.json', 'r') as f:
            return json.load(f)
    except:
        return []

def save_portfolio(items):
    with open('data/portfolio.json', 'w') as f:
        json.dump(items, f, ensure_ascii=False, indent=2)

@app.route('/admin/portfolio', methods=['GET', 'POST'])
def admin_portfolio():
    items = load_portfolio()
    if request.method == 'POST':
        new_item = {
            "title": request.form['title'],
            "description": request.form['description'],
            "image": "uploads/" + request.files['image'].filename,
            "tags": [tag.strip() for tag in request.form['tags'].split(',')]
        }
        items.append(new_item)
        save_portfolio(items)
        # Rasmni saqlash
        request.files['image'].save('static/' + new_item['image'])
        flash("Loyiha qo'shildi!", "success")
        return redirect(url_for('admin_portfolio'))
    return render_template('admin/portfolio.html', items=items)
