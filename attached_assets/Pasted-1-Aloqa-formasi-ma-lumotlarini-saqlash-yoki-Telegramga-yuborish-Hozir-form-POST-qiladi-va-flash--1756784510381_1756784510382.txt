# app.py da (contact uchun)
import requests

TELEGRAM_BOT_TOKEN = "sizning_bot_tokeningiz"
TELEGRAM_CHAT_ID = "sizning_chat_id"

@app.route('/contact', methods=['GET', 'POST'])
def contact():
    if request.method == 'POST':
        name = request.form['name']
        phone = request.form['phone']
        message = request.form['message']
        
        # Telegramga yuborish
        text = f"📝 Yangi murojaat!\n\n👤 Ism: {name}\n📞 Telefon: {phone}\n💬 Xabar: {message}"
        url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"
        requests.post(url, data={'chat_id': TELEGRAM_CHAT_ID, 'text': text})
        
        flash("Xabaringiz muvaffaqiyatli yuborildi! Tez orada aloqaga chiqamiz.", "success")
        return redirect(url_for('contact'))
    return render_template('contact.html')
