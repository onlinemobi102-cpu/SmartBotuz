import requests
from datetime import datetime

@app.route('/admin/ai/marketing/run', methods=['POST'])
def ai_daily_marketing():
    # 1. Trendlarni olish (Google Trends, RSS, Telegram)
    trends = get_trends()  # Masalan: "AI in education", "Telegram bot for business"

    # 2. Har bir trend uchun blog post yaratish
    posts = []
    for trend in trends[:5]:
        prompt = f"""
        O'zbek tilida SEO optimallashtirilgan blog maqolasini yarat:
        Mavzu: {trend}
        Uzunlik: 600 so'z
        Sarlavha: jozibador
        Paragraflar: 5-6 ta
        Kalit so'zlar: {trend}, aqlli yechimlar, avtomatlashtirish
        Oxirida: "SmartBot.uz â€” aqlli yechimlar, zamonaviy biznes uchun"
        """
        response = model.generate_content(prompt)
        post = {
            "title": extract_title(response.text),
            "content": response.text,
            "slug": trend.lower().replace(" ", "-"),
            "date": datetime.now().strftime("%Y-%m-%d")
        }
        posts.append(post)

        # 3. Blogga saqlash
        save_blog_post(post)

        # 4. Telegram kanaliga yuborish
        send_to_telegram_channel(post['title'], post['content'])

    # 5. Ma'lumotlarni qaytarish
    return {'posts': posts, 'count': len(posts)}
