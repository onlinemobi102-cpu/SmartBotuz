@app.route('/ai/blog', methods=['GET', 'POST'])
def ai_blog():
    if request.method == 'POST':
        topic = request.form['topic']
        # Gemini bilan maqola yaratish
        response = model.generate_content(f"O'zbek tilida SEO optimallashtirilgan blog maqolasini yarat: {topic}")
        return render_template('ai/blog_result.html', article=response.text)
    return render_template('ai/blog.html')

@app.route('/ai/case-study', methods=['POST'])
def ai_case_study():
    project_info = request.form['project_info']
    response = model.generate_content(f"O'zbek tilida batafsil case study yarat: {project_info}")
    return render_template('ai/case_study_result.html', case_study=response.text)

@app.route('/ai/analyze', methods=['POST'])
def ai_analyze():
    message = request.form['message']
    response = model.generate_content(f"O'zbek tilida mijoz xabarini tahlil qiling: {message}")
    return render_template('ai/analyze_result.html', analysis=response.text)
