# save_message(name, phone, message)
def save_message(data):
    messages = load_json('data/messages.json')
    messages.append({
        "id": len(messages) + 1,
        "name": data['name'],
        "phone": data['phone'],
        "message": data['message'],
        "date": datetime.now().isoformat()
    })
    save_json('data/messages.json', messages)
